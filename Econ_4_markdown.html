#Equation 1: Temperature over time
temp <- function(t,T){
  tau <- min(T*t/100,T)
  return(tau)
}

##-----------------------
#Equation 2: Activity retained over time 
K <- function(temp){
  output <- exp(-beta*(temp^2))
  return(output)
}

##-----------------------
#Equation 3: Total consumption over time
C <- function(K,t){
  output <- K*exp(g*t)
  return(output)
}

#-------------------------------
#Equation 4: Society's utility 
U <- function(C){
  output <- (C^(1-eta)/(1-eta))
  return(output)
}
